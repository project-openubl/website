import useDocusaurusContext from "@docusaurus/useDocusaurusContext";
import { RedocStandalone } from "redoc";

export const ApiReference = () => {
  const { siteConfig } = useDocusaurusContext();
  const redoConfig = {
    theme: {
      colors: {
        primary: {
          main: "#2b9af3",
        },
      },
      typography: {
        fontSize: "var(--ifm-font-size-base)",
        lineHeight: "var(--ifm-line-height-base)",
        fontFamily: "var(--ifm-font-family-base)",
        headings: {
          fontFamily: "var(--ifm-font-family-base)",
          fontWeight: "var(--ifm-heading-font-weight)",
        },
        code: {
          lineHeight: "var(--ifm-pre-line-height)",
          fontFamily: "var(--ifm-font-family-monospace)",
        },
      },
      sidebar: {
        width: "300px",
      },
    },
  };
  React.useEffect(() => {
    const wrapper = document.getElementsByClassName(
      "container padding-vert--lg"
    )[0];
    if (wrapper) {
      wrapper.classList.add("redoc-api-wrapper");
      wrapper.children[0].children[0].children[0].classList.add(
        "redoc-api-wrapper-third-child"
      );
    }
    return () => {
      if (wrapper) {
        wrapper.classList.remove("redoc-api-wrapper");
        wrapper.children[0].children[0].children[0].classList.remove(
          "redoc-api-wrapper-third-child"
        );
      }
    };
  }, []);
  return (
    <div>
      <RedocStandalone
        spec="/openapi/searchpe.yaml"
        options={redoConfig}
      />
    </div>
  );
};

<ApiReference />
